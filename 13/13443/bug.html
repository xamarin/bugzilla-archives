This statement, when I use it in a Xam.iOS app, crashes the compiler

switch (ev.ButtonIndex)
{
  case 0:
    (Invoice.Status == InvoiceStatus.Draft ? MarkAsSent() : MarkAsDraft());
    break;
}

	{
		public string s = "";
		public void Die()

			(s == "hello" ? MarkAsSent() : MarkAsDraft());
		}
		void MarkAsSent () {
		}
		void MarkAsDraft() {

		}
	}
